```rust
// 这是一种在稳定版 Rust（从 rustc 1.33 起）中模拟“带生命周期参数的 GAT（泛型关联类型）”的技巧。
//
// 我没有见过完全一致的实现方式，但如果说没人想到过，我也会很意外。
// 我认为它规避了我见过的其他生命周期 GATs 变通方案中的大多数缺点。
//
// 尤其是：无论是实现还是使用这种模拟生命周期 GATs 的 trait，都不需要额外定义辅助项。
// 只有在定义 trait 时，才需要为每个 GAT 增加一个辅助 trait（第二种变体还需要一个辅助 impl）。
// 这也意味着即使不写任何减少样板代码的宏，这种技巧依然可行。
//
// 以下是一个完整的示例，解释了这种技术的两种变体，并用常认为是 GATs 动机的 lending/streaming 迭代器 trait来说明。

// 先定义一个 trait：用于表示“对生命周期泛化后的 item 类型”。
// 如果需要给 `Item` 增加额外约束，也是在这里加。（本例仅使用隐式的 `Sized` 约束。）
pub trait GenericItem<'a> {
    type Item;
}

// 在用它定义 trait 时，不再直接写 `Item` 关联类型（那本应是一个 GAT），
// 而是改成一个关联类型 `GenericItem`。借助 HRTB，我们约束 `GenericItem`
// 能在任意给定生命周期下提供对应的 `Item` 类型。`?Sized` 的原因会在下文解释。
pub trait LendingIterator {
    type GenericItem: ?Sized + for<'any> GenericItem<'any>;

    // `&mut self` 和 `as GenericItem` 上的生命周期是省略写法（即同一个生命周期）。
    // 这意味着 `self` 的生命周期会作为参数传给 `Self::GenericItem` 上的 `GenericItem` trait，
    // 从而得到带有正确生命周期的关联类型 `Item`。
    fn next(&mut self) -> Option<<Self::GenericItem as GenericItem<'_>>::Item>;
}

// 接下来实现一个迭代器：在同一块可变切片上迭代“彼此重叠的可变子切片”。

pub struct WindowsMut<'a, T> {
    slice: &'a mut [T],
    offset: usize,
    size: usize,
}

impl<'a, T> WindowsMut<'a, T> {
    pub fn new(slice: &'a mut [T], size: usize) -> Self {
        Self {
            slice,
            offset: 0,
            size,
        }
    }
}

// 实现该 trait 时需要提供 `GenericItem`。
// 与其新建一个类型并为它实现 trait，不如直接用 trait object 类型。
// 这样可以内联写 HRTB，无需额外定义新类型，也不用真正去实现该 trait。
// 由于 trait object 是 DST（unsized 类型），上面才需要 `?Sized` 约束。
impl<'a, T> LendingIterator for WindowsMut<'a, T> {
    type GenericItem = dyn for<'any> GenericItem<'any, Item = &'any mut [T]>;

    // 在实现里直接使用非泛型写法，可读性会更好
    fn next(&mut self) -> Option<&mut [T]> {
        let offset = self.offset;
        self.offset += 1;
        self.slice.get_mut(offset..offset + self.size)
    }
}

// 另一种做法：使用函数指针类型。
// 只要先提供下面这个对应的覆盖实现即可：
impl<'a, Item, T> GenericItem<'a> for T
where
    T: FnOnce(&'a ()) -> Item,
{
    type Item = Item;
}

// 再实现另一个迭代器。
pub struct PrefixesMut<'a, T> {
    slice: &'a mut [T],
    size: usize,
}

impl<'a, T> PrefixesMut<'a, T> {
    pub fn new(slice: &'a mut [T]) -> Self {
        Self { slice, size: 0 }
    }
}

// 借助 `FnOnce` 的 blanket impl，就可以在 trait 实现里使用生命周期省略写法，
// 以及 `fn` 语法自带的隐式 HRTB。
// 相比 `dyn for<'any> GenericItem<'any, Item = ...`，我觉得这种写法更易读。
impl<'a, T> LendingIterator for PrefixesMut<'a, T> {
    type GenericItem = fn(&()) -> &mut [T];

    fn next(&mut self) -> Option<&mut [T]> {
        self.size += 1;
        self.slice.get_mut(..self.size)
    }
}

// 使用这些 trait 实现时，也不需要写任何特殊语法。
pub fn main() {
    let mut numbers = vec![0; 20];

    let mut windows = WindowsMut::new(&mut numbers, 5);

    while let Some(window) = windows.next() {
        for x in window {
            *x += 1;
        }
    }

    println!("{:?}", numbers);

    let mut chunk_prefixes = PrefixesMut::new(&mut numbers);

    while let Some(prefix) = chunk_prefixes.next() {
        prefix.swap(prefix.len() / 2, prefix.len() - 1);
    }

    println!("{:?}", numbers);
}
```
